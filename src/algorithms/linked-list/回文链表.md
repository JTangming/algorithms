### 回文链表

原题：https://leetcode-cn.com/problems/palindrome-linked-list-lcci/

#### 递归
```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {boolean}
 */
var isPalindrome = function(head) {
    let left = head;
    const reverse = (right) => {
        if (right === null) return true;
        let res = reverse(right.next);
        res = res && (left.val === right.val)
        left = left.next;

        return res;
    }

    return reverse(head);
};
```

#### 双指针

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {boolean}
 */
var isPalindrome = function(head) {
    let slow = fast = left = head;
    // 最为关键的是找到中间节点，分两种情况：
    // ① 奇数个，通过快慢指针遍历，fast 指向 null，则 slow 为中间节点
    // ② 偶数个，有两个中间节点，我们需要第二个中间节点
    while (fast !== null && fast.next !== null) {
        slow = slow.next;
        fast = fast.next.next;
    }

    // 奇数个节点，可以少处理一个节点。
    if (fast !== null) slow = slow.next;

    let right = reverseList(slow);

    while (right !== null) {
        if (right.val !== left.val) {
            return false;
        }
        right = right.next;
        left = left.next;
    }

    return true;
};

// 反转链表（老掉牙的东西了）
function reverseList(head) {
    let cur = head, prev = null;
    while (cur !== null) {
        let temp = cur.next;
        cur.next = prev;
        prev = cur;
        cur = temp;
    }

    return prev;
}
```
