/********** 堆排序 */
function heapSort (nums) { // 输入 [5,2,3,10,8,1,4]
    let len = nums.length;
    let heapArr = new Array(len + 1);

    for (let i = 0; i < len; i++) {
        heapArr[i + 1] = nums[i];
    }

    // 对原数组进行堆化，采用下沉操作，把大的元素浮动上来，建立一个大堆顶
    // 从最后一个非叶子节点开始
    for (let j = Math.floor(len/2); j > 0; j--) {
        heapify(heapArr, j, len);
    }
    console.log(heapArr); // [ <1 empty item>, 10, 8, 4, 2, 5, 1, 3 ]

    let k = len;
    // 基于以上的大堆顶进行排序
    while (k > 1) {
        // 把最大的第一个元素和最后一个叶子节点对换，通过迭代的方式依次对前一个节点进行类似的操作，直到第一个元素为止完成排序
        swap(heapArr, 1, k--);
        // 交换元素后，从数组第一个元素开始堆化，目的是把大的元素上浮，重新建立一个大堆顶
        heapify(heapArr, 1, k);
    }
    for (let i = 1; i < len + 1; ++i) {
        nums[i-1] = heapArr[i];
    }
    return nums;
}

// 堆化
function heapify(arr, k, len) {
    while (2 * k <= len) {
        let i = 2 * k;
        if (i + 1 <= len && arr[i] < arr[i + 1]) {
            i++;
        }
        if (arr[k] < arr[i]) {
            swap(arr, k, i); 
        } else break;
        k = i;
    }
}

function swap(arr, i, j) {
    let temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
}

/********** 希尔排序 */
function shellSort(nums) {
    let increment = nums.length;
    while (increment > 1) {
        increment = Math.floor(increment / 2);
        // 插入排序的变形
        for (let i = increment; i < nums.length; i++) {
            let j, temp = nums[i];
            for (j = i - increment; j >= 0; j -= increment) {
                if (temp < nums[j]) {
                    nums[j + increment] = nums[j];
                    continue;
                }
                break;
            }
            nums[j + increment] = temp;
        }
    }
    return nums;
}

/********** 插入排序 */
function inertSort(nums) {
    let len = nums.length;
    // 从 1 开始
    for (let i = 1; i < len; i++) {
        let j, temp = nums[i]; // 记录当前位置的值
        for (j = i - 1; j >= 0; j--) {
            // 如果已排序的元素 j 的值大于当前值 temp，则当前元素置换一下
            if (nums[j] > temp) {
                nums[j + 1] = nums[j];
                continue;
            }
            // 否则跳出循环，表示当前位置正确的
            break;
        }
        nums[j + 1] = temp;
    }
    return nums;
}

/********** 快速排序 */
function quickSort(arr, left, right) {
    if (left < right) {
        let pos = partition(arr, left, right);
        quickSort(arr, left, pos - 1);
        quickSort(arr, pos + 1, right);
    }

    return arr;
}

function partition(arr, left ,right) {
    let pivot = right;
    let index = left;
    for(let i = index; i < right; i++) {
        if (arr[i] < arr[pivot]) {
            swap(arr, index, i);
            index++;
        }
    }
    swap(arr, index, pivot);

    return index;
}

function swap(arr, i, j) {
    let temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
}
